List rev(List root)
{
	decl
		List temp;
	enddecl
	
	begin
		if(root.next==null) then
			head=root;
			
		else
			temp=rev(root.next);
			temp.next=root;
			root.next=null;
			
		endif;
		return root;
	end
}
